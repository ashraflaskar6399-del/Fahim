import { useState, useEffect } from "react";

const PENALTY_POOL = [
  "Complete DOUBLE your normal workout — no excuses, Hunter.",
  "Run 5km penalty sprint before anything else.",
  "100 pushups + 100 situps + 100 squats. Begin.",
  "4-hour deep focus study session — phone locked.",
  "Complete all missed quests + 50% bonus reps.",
  "Cold shower + 2x full body workout. Weakness is unacceptable.",
];

const TODAY = () => new Date().toISOString().split("T")[0];

const formatDate = (d) => {
  const dt = new Date(d);
  return `${String(dt.getMonth() + 1).padStart(2, "0")}/${String(dt.getDate()).padStart(2, "0")}/${dt.getFullYear()}`;
};

const loadData = async (key, def) => {
  try {
    const r = await window.storage.get(key);
    return r ? JSON.parse(r.value) : def;
  } catch {
    return def;
  }
};

const saveData = async (key, val) => {
  try {
    await window.storage.set(key, JSON.stringify(val));
  } catch {}
};

const DEFAULT_LISTS = [
  {
    id: "default",
    name: "⚔️ Default Quests",
    quests: [
      { id: "d1", text: "Drink 2L of water", completed: false, completedDate: null },
      { id: "d2", text: "Sleep before midnight", completed: false, completedDate: null },
    ],
  },
  {
    id: "workout",
    name: "💪 Today Workout Quests",
    quests: [
      { id: "w1", text: "20 Pushups", completed: false, completedDate: null },
      { id: "w2", text: "20 Situps", completed: false, completedDate: null },
      { id: "w3", text: "20 Squats", completed: false, completedDate: null },
      { id: "w4", text: "5km Run", completed: false, completedDate: null },
    ],
  },
  {
    id: "study",
    name: "📚 Today Study Quests",
    quests: [],
  },
];

const getRank = (lv) => {
  if (lv < 5) return { rank: "E", label: "E-Rank Hunter", color: "#94a3b8" };
  if (lv < 10) return { rank: "D", label: "D-Rank Hunter", color: "#4ade80" };
  if (lv < 15) return { rank: "C", label: "C-Rank Hunter", color: "#60a5fa" };
  if (lv < 20) return { rank: "B", label: "B-Rank Hunter", color: "#a78bfa" };
  if (lv < 30) return { rank: "A", label: "A-Rank Hunter", color: "#f97316" };
  return { rank: "S", label: "S-Rank Hunter", color: "#fbbf24" };
};

const getXPNeeded = (lv) => lv * 100;

export default function SoloLevelingApp() {
  const [lists, setLists] = useState(null);
  const [finished, setFinished] = useState([]);
  const [checkIns, setCheckIns] = useState({});
  const [lv, setLv] = useState({ level: 1, xp: 0, totalDays: 0 });
  const [penalties, setPenalties] = useState([]);
  const [selectedList, setSelectedList] = useState("workout");
  const [view, setView] = useState("quests");
  const [moreOpen, setMoreOpen] = useState(false);
  const [newQuestText, setNewQuestText] = useState("");
  const [newListName, setNewListName] = useState("");
  const [showAddList, setShowAddList] = useState(false);
  const [loaded, setLoaded] = useState(false);
  const [notif, setNotif] = useState(null);
  const [levelUpAnim, setLevelUpAnim] = useState(false);
  const [editingListId, setEditingListId] = useState(null);
  const [editListName, setEditListName] = useState("");
  const [editPenaltyId, setEditPenaltyId] = useState(null);
  const [editPenaltyText, setEditPenaltyText] = useState("");

  useEffect(() => {
    (async () => {
      const ql = await loadData("sl-lists", DEFAULT_LISTS);
      const fin = await loadData("sl-finished", []);
      const ci = await loadData("sl-checkins", {});
      const lvd = await loadData("sl-level", { level: 1, xp: 0, totalDays: 0 });
      const pen = await loadData("sl-penalties", []);
      setLists(ql);
      setFinished(fin);
      setCheckIns(ci);
      setLv(lvd);
      setPenalties(pen);
      setLoaded(true);
    })();
  }, []);

  // Auto-check penalty for yesterday
  useEffect(() => {
    if (!loaded) return;
    const yest = new Date();
    yest.setDate(yest.getDate() - 1);
    const yStr = yest.toISOString().split("T")[0];
    if (checkIns[yStr] === false && !penalties.some((p) => p.missedDate === yStr)) {
      addPenaltyForDate(yStr, penalties);
    }
  }, [loaded]);

  const addPenaltyForDate = (date, currentPens) => {
    const pen = {
      id: Date.now().toString(),
      missedDate: date,
      quest: PENALTY_POOL[Math.floor(Math.random() * PENALTY_POOL.length)],
      completed: false,
    };
    const newPens = [...currentPens, pen];
    setPenalties(newPens);
    saveData("sl-penalties", newPens);
  };

  const showNotif = (msg, type = "info") => {
    setNotif({ msg, type });
    setTimeout(() => setNotif(null), 3500);
  };

  const completeQuest = (listId, questId) => {
    const today = TODAY();
    const list = lists.find((l) => l.id === listId);
    const quest = list?.quests.find((q) => q.id === questId);
    if (!quest) return;

    const newLists = lists.map((l) =>
      l.id !== listId ? l : { ...l, quests: l.quests.filter((q) => q.id !== questId) }
    );
    const completedQ = { ...quest, completed: true, completedDate: today, fromList: listId, fromListName: list.name };
    const newFinished = [...finished, completedQ];

    let xpGain = 10;
    let newCheckIns = { ...checkIns };
    let newTotalDays = lv.totalDays;
    let checkedInToday = false;

    if (!newCheckIns[today]) {
      newCheckIns[today] = true;
      newTotalDays += 1;
      xpGain += 50;
      checkedInToday = true;
    }

    let newXP = lv.xp + xpGain;
    let newLevel = lv.level;
    let didLevelUp = false;

    while (newXP >= getXPNeeded(newLevel)) {
      newXP -= getXPNeeded(newLevel);
      newLevel += 1;
      didLevelUp = true;
    }

    const newLvData = { level: newLevel, xp: newXP, totalDays: newTotalDays };

    setLists(newLists);
    setFinished(newFinished);
    setCheckIns(newCheckIns);
    setLv(newLvData);
    saveData("sl-lists", newLists);
    saveData("sl-finished", newFinished);
    saveData("sl-checkins", newCheckIns);
    saveData("sl-level", newLvData);

    if (didLevelUp) {
      setLevelUpAnim(true);
      setTimeout(() => setLevelUpAnim(false), 3000);
      showNotif(`🎉 LEVEL UP! You are now LEVEL ${newLevel}!`, "levelup");
    } else if (checkedInToday) {
      showNotif(`✔ Quest Complete! +60 XP | Check-In Recorded for ${formatDate(today)}`, "success");
    } else {
      showNotif(`✔ Quest Complete! +10 XP`, "success");
    }
  };

  const addQuest = () => {
    if (!newQuestText.trim()) return;
    const q = { id: Date.now().toString(), text: newQuestText.trim(), completed: false, completedDate: null };
    const newLists = lists.map((l) => l.id === selectedList ? { ...l, quests: [...l.quests, q] } : l);
    setLists(newLists);
    saveData("sl-lists", newLists);
    setNewQuestText("");
  };

  const deleteQuest = (listId, questId) => {
    const newLists = lists.map((l) => l.id !== listId ? l : { ...l, quests: l.quests.filter((q) => q.id !== questId) });
    setLists(newLists);
    saveData("sl-lists", newLists);
  };

  const addList = () => {
    if (!newListName.trim()) return;
    const list = { id: Date.now().toString(), name: newListName.trim(), quests: [] };
    const newLists = [...lists, list];
    setLists(newLists);
    saveData("sl-lists", newLists);
    setNewListName("");
    setShowAddList(false);
    setSelectedList(list.id);
    setView("quests");
  };

  const deleteList = (listId) => {
    if (listId === "default") return;
    const newLists = lists.filter((l) => l.id !== listId);
    setLists(newLists);
    saveData("sl-lists", newLists);
    setSelectedList(newLists[0]?.id || "default");
  };

  const saveListName = (listId) => {
    if (!editListName.trim()) { setEditingListId(null); return; }
    const newLists = lists.map((l) => l.id === listId ? { ...l, name: editListName.trim() } : l);
    setLists(newLists);
    saveData("sl-lists", newLists);
    setEditingListId(null);
  };

  const completePenalty = (penId) => {
    const newPens = penalties.map((p) => p.id === penId ? { ...p, completed: true } : p);
    setPenalties(newPens);
    saveData("sl-penalties", newPens);
    const newXP = lv.xp + 100;
    const newLvData = { ...lv, xp: newXP };
    setLv(newLvData);
    saveData("sl-level", newLvData);
    showNotif("⚡ Penalty Complete! +100 XP — XP Unlocked!", "success");
  };

  const savePenaltyEdit = (penId) => {
    if (!editPenaltyText.trim()) { setEditPenaltyId(null); return; }
    const newPens = penalties.map((p) => p.id === penId ? { ...p, quest: editPenaltyText.trim() } : p);
    setPenalties(newPens);
    saveData("sl-penalties", newPens);
    setEditPenaltyId(null);
  };

  const getHistory = () => {
    const days = [];
    const today = new Date();
    for (let i = 29; i >= 0; i--) {
      const d = new Date(today);
      d.setDate(d.getDate() - i);
      const dStr = d.toISOString().split("T")[0];
      days.push({
        date: dStr,
        label: `D${30 - i}`,
        checked: checkIns[dStr] === true,
        isToday: i === 0,
        isPast: i > 0 && checkIns[dStr] === undefined,
      });
    }
    return days;
  };

  if (!loaded) {
    return (
      <div style={S.app}>
        <div style={S.bootScreen}>
          <div style={S.bootLogo}>⬡</div>
          <div style={S.bootText}>SYSTEM INITIALIZING...</div>
          <div style={S.bootBar}><div style={S.bootBarFill}></div></div>
        </div>
        <style>{CSS}</style>
      </div>
    );
  }

  const curList = lists?.find((l) => l.id === selectedList);
  const history = getHistory();
  const pendingPens = penalties.filter((p) => !p.completed);
  const xpPct = Math.min(100, Math.round((lv.xp / getXPNeeded(lv.level)) * 100));
  const rankInfo = getRank(lv.level);

  return (
    <div style={S.app} onClick={() => moreOpen && setMoreOpen(false)}>
      <style>{CSS}</style>

      {/* Ambient particles */}
      <div style={S.particles}>
        {[...Array(6)].map((_, i) => <div key={i} className={`particle particle-${i}`} />)}
      </div>

      {/* Level Up Overlay */}
      {levelUpAnim && (
        <div style={S.levelUpOverlay}>
          <div style={S.levelUpBox}>
            <div style={S.levelUpStar}>★</div>
            <div style={S.levelUpTitle}>LEVEL UP</div>
            <div style={S.levelUpNum}>LV. {lv.level}</div>
            <div style={S.levelUpRank}>{rankInfo.label}</div>
          </div>
        </div>
      )}

      {/* Notification */}
      {notif && (
        <div className="notif-slide" style={{ ...S.notif, ...(notif.type === "levelup" ? S.notifLevelup : notif.type === "success" ? S.notifSuccess : {}) }}>
          {notif.msg}
        </div>
      )}

      {/* Header */}
      <div style={S.header}>
        <div style={S.headerLeft}>
          <div style={S.hexIcon}>⬡</div>
          <div>
            <div style={S.sysLabel}>SYSTEM INTERFACE</div>
            <div style={S.appTitle}>SOLO QUEST</div>
          </div>
        </div>
        <div style={S.headerRight}>
          <div style={S.xpMiniBar}>
            <div style={{ ...S.xpMiniFill, width: `${xpPct}%` }} />
          </div>
          <div style={{ ...S.rankBadge, borderColor: rankInfo.color, color: rankInfo.color }}>
            {rankInfo.rank}
          </div>
          <div style={S.lvBadge}>LV.{lv.level}</div>
          <button style={S.moreBtn} onClick={(e) => { e.stopPropagation(); setMoreOpen(!moreOpen); }} className="more-btn">
            {moreOpen ? "✕" : "☰"}
          </button>
        </div>
      </div>

      {/* Dropdown More Menu */}
      {moreOpen && (
        <div style={S.moreMenu} onClick={(e) => e.stopPropagation()}>
          <div style={S.moreMenuTitle}>── SYSTEM MENU ──</div>
          <button style={S.moreItem} className="more-item" onClick={() => { setView("checkin"); setMoreOpen(false); }}>
            <span style={S.moreIcon}>📅</span>
            <div>
              <div style={S.moreItemLabel}>Daily Check-In</div>
              <div style={S.moreItemSub}>Total: {lv.totalDays} days active</div>
            </div>
          </button>
          <button style={S.moreItem} className="more-item" onClick={() => { setView("level"); setMoreOpen(false); }}>
            <span style={S.moreIcon}>⚡</span>
            <div>
              <div style={S.moreItemLabel}>Level Progress</div>
              <div style={S.moreItemSub}>{rankInfo.label} · {xpPct}% to next level</div>
            </div>
          </button>
          <button
            style={{ ...S.moreItem, ...(pendingPens.length > 0 ? S.moreItemPenalty : {}) }}
            className="more-item"
            onClick={() => { setView("penalties"); setMoreOpen(false); }}>
            <span style={S.moreIcon}>💀</span>
            <div>
              <div style={S.moreItemLabel}>Penalty Quests {pendingPens.length > 0 ? `(${pendingPens.length} ACTIVE)` : ""}</div>
              <div style={S.moreItemSub}>{pendingPens.length > 0 ? "⚠ XP LOCKED — Complete penalties" : "No active penalties"}</div>
            </div>
          </button>
        </div>
      )}

      {/* Body */}
      <div style={S.body}>
        {/* Sidebar */}
        <div style={S.sidebar}>
          <div style={S.sidebarHead}>QUEST LISTS</div>

          {lists?.map((list) => (
            <div key={list.id}>
              {editingListId === list.id ? (
                <div style={S.editListRow}>
                  <input
                    style={S.editListInput}
                    value={editListName}
                    onChange={(e) => setEditListName(e.target.value)}
                    onKeyDown={(e) => { if (e.key === "Enter") saveListName(list.id); if (e.key === "Escape") setEditingListId(null); }}
                    autoFocus
                  />
                  <button style={S.editSaveBtn} onClick={() => saveListName(list.id)}>✔</button>
                </div>
              ) : (
                <div
                  style={{ ...S.listItem, ...(selectedList === list.id && view === "quests" ? S.listItemActive : {}) }}
                  className="list-item"
                  onClick={() => { setSelectedList(list.id); setView("quests"); }}>
                  <span style={S.listItemText}>{list.name}</span>
                  <div style={S.listItemRight}>
                    <span style={S.listCount}>{list.quests.length}</span>
                    <button style={S.listEditBtn} className="list-edit-btn"
                      onClick={(e) => { e.stopPropagation(); setEditingListId(list.id); setEditListName(list.name); }}>✎</button>
                    {list.id !== "default" && (
                      <button style={S.listDelBtn} className="list-edit-btn"
                        onClick={(e) => { e.stopPropagation(); deleteList(list.id); }}>✕</button>
                    )}
                  </div>
                </div>
              )}
            </div>
          ))}

          <div
            style={{ ...S.listItem, ...(view === "finished" ? S.listItemActive : {}) }}
            className="list-item"
            onClick={() => setView("finished")}>
            <span style={S.listItemText}>✅ Finished Quests</span>
            <span style={S.listCount}>{finished.length}</span>
          </div>

          <div style={S.sidebarDivider} />

          {showAddList ? (
            <div style={S.addListForm}>
              <input
                style={S.addListInput}
                value={newListName}
                onChange={(e) => setNewListName(e.target.value)}
                placeholder="Quest list name..."
                onKeyDown={(e) => { if (e.key === "Enter") addList(); if (e.key === "Escape") setShowAddList(false); }}
                autoFocus
              />
              <div style={{ display: "flex", gap: 4 }}>
                <button style={S.addListConfirm} onClick={addList}>CREATE</button>
                <button style={S.addListCancel} onClick={() => setShowAddList(false)}>✕</button>
              </div>
            </div>
          ) : (
            <button style={S.newListBtn} className="new-list-btn" onClick={() => setShowAddList(true)}>
              + New Quest List
            </button>
          )}
        </div>

        {/* Main Panel */}
        <div style={S.panel}>

          {/* QUESTS VIEW */}
          {view === "quests" && curList && (
            <div style={S.panelInner}>
              <div style={S.panelHeader}>
                <div style={S.panelCorner} />
                <div style={S.panelTitle}>{curList.name}</div>
                <div style={S.panelSubtitle}>{curList.quests.length} QUEST{curList.quests.length !== 1 ? "S" : ""} ASSIGNED</div>
              </div>

              <div style={S.questList}>
                {curList.quests.length === 0 && (
                  <div style={S.empty}>
                    <div style={S.emptyIcon}>⚔</div>
                    <div style={S.emptyTitle}>NO QUESTS ASSIGNED</div>
                    <div style={S.emptySub}>Add a quest below to begin your journey.</div>
                  </div>
                )}
                {curList.quests.map((q, i) => (
                  <div key={q.id} style={S.questRow} className="quest-row">
                    <div style={S.questNum}>{String(i + 1).padStart(2, "0")}</div>
                    <button style={S.checkBtn} className="check-btn" onClick={() => completeQuest(curList.id, q.id)}>
                      <div style={S.checkInner} />
                    </button>
                    <span style={S.questLabel}>{q.text}</span>
                    <button style={S.delQBtn} className="del-q-btn" onClick={() => deleteQuest(curList.id, q.id)}>✕</button>
                  </div>
                ))}
              </div>

              <div style={S.addQuestRow}>
                <div style={S.addQuestInputWrap}>
                  <span style={S.addQuestPlus}>▷</span>
                  <input
                    style={S.addQuestInput}
                    value={newQuestText}
                    onChange={(e) => setNewQuestText(e.target.value)}
                    placeholder="Enter new quest..."
                    onKeyDown={(e) => e.key === "Enter" && addQuest()}
                  />
                </div>
                <button style={S.addQuestBtn} className="add-quest-btn" onClick={addQuest}>＋ ADD QUEST</button>
              </div>
            </div>
          )}

          {/* FINISHED VIEW */}
          {view === "finished" && (
            <div style={S.panelInner}>
              <div style={S.panelHeader}>
                <div style={S.panelCorner} />
                <div style={S.panelTitle}>✅ Finished Quests</div>
                <div style={S.panelSubtitle}>{finished.length} COMPLETED</div>
              </div>
              <div style={S.questList}>
                {finished.length === 0 && (
                  <div style={S.empty}>
                    <div style={S.emptyIcon}>🏆</div>
                    <div style={S.emptyTitle}>NO COMPLETED QUESTS YET</div>
                    <div style={S.emptySub}>Complete quests to see them here.</div>
                  </div>
                )}
                {[...finished].reverse().map((q, i) => (
                  <div key={q.id + i} style={{ ...S.questRow, opacity: 0.75 }}>
                    <span style={S.completedCheck}>✔</span>
                    <div style={{ flex: 1 }}>
                      <div style={{ ...S.questLabel, textDecoration: "line-through", color: "#6b8fa8" }}>{q.text}</div>
                      <div style={S.completedMeta}>{q.fromListName} · {formatDate(q.completedDate)}</div>
                    </div>
                  </div>
                ))}
              </div>
            </div>
          )}

          {/* CHECK-IN VIEW */}
          {view === "checkin" && (
            <div style={S.panelInner}>
              <div style={S.panelHeader}>
                <div style={S.panelCorner} />
                <div style={S.panelTitle}>📅 Daily Check-In</div>
                <div style={S.panelSubtitle}>CHECK-IN HISTORY · LAST 30 DAYS</div>
              </div>
              <div style={S.streakRow}>
                <div style={S.streakCard}>
                  <div style={S.streakNum}>{lv.totalDays}</div>
                  <div style={S.streakLabel}>DAYS ACTIVE</div>
                </div>
                <div style={S.streakCard}>
                  <div style={{ ...S.streakNum, color: "#f97316" }}>{pendingPens.length}</div>
                  <div style={S.streakLabel}>PENALTIES</div>
                </div>
                <div style={S.streakCard}>
                  <div style={{ ...S.streakNum, color: rankInfo.color }}>{lv.level}</div>
                  <div style={S.streakLabel}>LEVEL</div>
                </div>
              </div>
              <div style={S.ciNote}>Check-in is automatically recorded when you complete at least one quest.</div>
              <div style={S.histGrid}>
                {history.map((day, i) => (
                  <div
                    key={day.date}
                    style={{
                      ...S.dayCell,
                      ...(day.checked ? S.dayCellOn : {}),
                      ...(day.isToday ? S.dayCellToday : {}),
                      ...(!day.checked && !day.isToday && !day.isPast ? S.dayCellMiss : {}),
                    }}>
                    <div style={S.dayLabel}>{day.label}</div>
                    <div style={S.dayIcon}>{day.checked ? "✔️" : day.isToday ? "◉" : day.isPast ? "–" : "❎"}</div>
                    <div style={S.dayDate}>{day.date.slice(5)}</div>
                  </div>
                ))}
              </div>
            </div>
          )}

          {/* LEVEL VIEW */}
          {view === "level" && (
            <div style={S.panelInner}>
              <div style={S.panelHeader}>
                <div style={S.panelCorner} />
                <div style={S.panelTitle}>⚡ Level Progress</div>
                <div style={S.panelSubtitle}>HUNTER STATUS BOARD</div>
              </div>
              <div style={S.profileCard}>
                <div style={{ ...S.rankCircle, borderColor: rankInfo.color, boxShadow: `0 0 24px ${rankInfo.color}55` }}>
                  <div style={{ ...S.rankLetter, color: rankInfo.color }}>{rankInfo.rank}</div>
                </div>
                <div style={S.profileInfo}>
                  <div style={{ ...S.profileRankLabel, color: rankInfo.color }}>{rankInfo.label}</div>
                  <div style={S.profileLevel}>LEVEL {lv.level}</div>
                  <div style={S.xpBarOuter}>
                    <div style={{ ...S.xpBarInner, width: `${xpPct}%`, background: rankInfo.color }} />
                    <div style={S.xpBarGlow} />
                  </div>
                  <div style={S.xpNums}>{lv.xp} / {getXPNeeded(lv.level)} XP — {xpPct}%</div>
                </div>
              </div>

              <div style={S.roadmapTitle}>── LEVELING ROADMAP ──</div>
              {[
                { range: "LV 1–5", label: "The 7-Day Sprint", desc: "7 days of daily check-ins per level", color: "#4ade80", current: lv.level <= 5 },
                { range: "LV 6–15", label: "The 15-Day Grind", desc: "15 days of daily check-ins per level", color: "#facc15", current: lv.level > 5 && lv.level <= 15 },
                { range: "LV 16+", label: "The 30-Day Mastery", desc: "30 days of daily check-ins per level", color: "#f97316", current: lv.level > 15 },
              ].map((r) => (
                <div key={r.range} style={{ ...S.roadItem, ...(r.current ? { ...S.roadItemActive, borderLeftColor: r.color } : {}) }}>
                  <div style={{ ...S.roadBadge, color: r.color, borderColor: r.color }}>{r.range}</div>
                  <div style={S.roadInfo}>
                    <div style={S.roadLabel}>{r.label} {r.current && <span style={S.roadCurrent}>◀ YOU ARE HERE</span>}</div>
                    <div style={S.roadDesc}>{r.desc}</div>
                  </div>
                </div>
              ))}

              <div style={S.xpGuide}>
                <div style={S.xpGuideTitle}>XP GUIDE</div>
                <div style={S.xpGuideRow}><span style={{ color: "#00d4ff" }}>Quest Complete</span><span>+10 XP</span></div>

          {view === "penalties" && (

              <div style={S.panelHeader}>

                  <div style={S.emptyTitle}>NO PENALTIES</div>
                  <div style={S.emptySub}>Stay consistent. Don't let this list grow.</div>
                </div>
              )}

              {penalties.map((p) => (
                <div key={p.id} style={{ ...S.penCard, ...(p.completed ? S.penCardDone : {}) }}>
                  <div style={S.penDateRow}>
                    <span style={S.penDate}>Missed: {formatDate(p.missedDate)}</span>
                    {!p.completed && (
                      <button style={S.penEditBtn} onClick={() => { setEditPenaltyId(p.id); setEditPenaltyText(p.quest); }}>✎ Edit</button>
                    )}
                  </div>
                  {editPenaltyId === p.id ? (
                    <div style={S.penEditRow}>
                      <input style={S.penEditInput} value={editPenaltyText}
                        onChange={(e) => setEditPenaltyText(e.target.value)}
                        onKeyDown={(e) => { if (e.key === "Enter") savePenaltyEdit(p.id); if (e.key === "Escape") setEditPenaltyId(null); }} autoFocus />
                      <button style={S.penEditSave} onClick={() => savePenaltyEdit(p.id)}>✔ Save</button>
                    </div>
                  ) : (
                    <div style={S.penQuest}>{p.quest}</div>
                  )}
                  {!p.completed ? (
                    <button style={S.penCompleteBtn} className="pen-complete-btn" onClick={() => completePenalty(p.id)}>
                      MARK COMPLETE ✔
                    </button>
                  ) : (
                    <div style={S.penDoneLabel}>✔ PENALTY CLEARED — XP Restored</div>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
